## 动态导入
## BigInt
```js
const max = 2 ** 53;
console.log(max);
console.log(Number.MAX_SAFE_INTEGER);

console.log(max === max + 1); // true 有最大值限制

// 第一种写法
const bigInt = 9007199254740993n;
console.log(bigInt);
console.log(typeof bigInt); // bigint

console.log(1n == 1); // true
console.log(1n === 1); // false

// 第二种写法
const bigInt2 = BigInt(9007199254740993n);
console.log(bigInt2);

const num = bigInt + bigInt2;
console.log(num.toString());
```

## Promise 扩展
```js
// 有一个失败，就只会返回失败的状态和结果，无法获取其他成功的请求状态和结果
Promise.all([
  Promise.resolve({
    code: 200,
    data: [1, 2, 3],
  }),
  Promise.reject({
    code: 500,
    data: [],
  }),
  Promise.resolve({
    code: 200,
    data: [7, 8, 9],
  }),
])
  .then((res) => {
    console.log(res);
    console.log("成功");
  })
  .catch((err) => {
    console.log(err);
    console.log("失败");
  });

// Promise.allSettled 即使失败也会返回相应的状态和结果
Promise.allSettled([
  Promise.resolve({
    code: 200,
    data: [1, 2, 3],
  }),
  Promise.reject({
    code: 500,
    data: [],
  }),
  Promise.resolve({
    code: 200,
    data: [7, 8, 9],
  }),
])
  .then((res) => {
    console.log(res);
    console.log("成功");
  })
  .catch((err) => {
    console.log(err);
    console.log("失败");
  });
```

## globalThis
提供了一个标准的方式去获取不同环境下的全局对象
```js
// node 中的全局对象 global
// web: window self(只窗口本身，同 window)
self.setTimeout(() => {
  console.log("es2020");
}, 1000);

// 之前获取全局对象
const getGlobal = () => {
  if (typeof self !== "undefined") {
    return self;
  }
  if (typeof window !== "undefined") {
    return window;
  }
  if (typeof global !== "undefined") {
    return global;
  }
  throw new Error("取法找到全局对象");
};
const global = getGlobal();
console.log(global);

// ES10 中
console.log(globalThis);
```

## 可选链
```js
// 可选链
const user = {
  address: {
    street: "xx街道",
    getNum() {
      return "80号";
    },
  },
};

// 之前
const street = user && user.address && user.address.street;
console.log(street);
const num =
  user && user.address && user.address.getNum && user.address.getNum();
console.log(num);

// ES1
const street = user?.address?.street;
console.log(street);
const num = user?.address?.getNum?.();
console.log(num);
```

## 空值合并运算符（Nullish coalescing Operator）
```js
const b = 0;
const a = b || 5;
console.log(a);

// ?? 只有前面是 null 或者 undefined 时才取后面的值
const b = "";
const a = b ?? 6;
console.log(a);
```
